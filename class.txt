public class Math {
	int x;
	int y;

	public Math(int x, int y) { // two-parameter constructor
		this.x = x;
		this.y = y;
	}

	public int Addition() {
		return x + y;
	}

	public int Subtraction() {
		return x - y
	}

	public int Multiplication() {
		return x * y;
	}

	public int Division() {
		return x / y;
	}
}
